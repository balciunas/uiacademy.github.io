phonecatApp.config(['$routeProvider',
  function($routeProvider) {
    $routeProvider.
      when('/person', {
        templateUrl: 'partials/person-list.html',
        controller: 'PersonListCtrl'
      }).
      when('/person/:id', {
        templateUrl: 'partials/person-detail.html',
        controller: 'PersonDetailCtrl'
      }).
      when('/person/:id/:work_id', {
        templateUrl: 'partials/work-detail.html',
        controller: 'WorkDetailCtrl'
      }).
      otherwise({
        redirectTo: '/person'
      });
  }]);

Since Underscore is a Backbone dependency and you already have it on your page, you can easily take advantage of its templating engine without adding any additional dependencies for your application.
